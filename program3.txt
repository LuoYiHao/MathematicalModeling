#include <stdio.h>
#include <math.h>
#include <iostream>
using namespace std;
double transformtoAngle = 180.0/3.1415926;
int main()
{
    double h[20001];
    h[0] = 0;
    for(int i = 1;i < 20001;i++)
    {
        h[i] = h[i - 1] + 0.0001;
    }

    for(int i = 0;i < 20001;i++)
    {
        for(int x = 20;x <= 25;x += 0.1)
        {
            for(int m = 1700; m <=1800;m++)
            {
                double wind = 0.625 * (4 - 2 * h[i]) * 36 * 36;
                //cout<<"h = "<<h[i];
                double water = 374 * 2 * h[i] * 36 * 36;
                double b0 = 1.025 * 1000 * 9.8 * h[i] * 3.1415926;
                double g0 = 1000 * 9.8;
                double t1 = sqrt(pow((b0 - g0),2) + pow((wind + water),2));
                double angle1 = atan((wind + water)/(b0 - g0));
                //cout<<" t1 = "<<t1<<" angle1 = "<<angle1 * transformtoAngle<<" ";

                double stream1 = 374 * cos(angle1) * 0.05 * 1.5 * 1.5;
                double t2 = sqrt(pow((19.723314 + t1 * cos(angle1) - 98),2) + pow((t1 * sin(angle1) + stream1),2));
                double angle2 = atan((t1 * sin(angle1) + stream1) / (19.723314 + t1 * cos(angle1) - 98));
                //cout<<"t2 = "<<t2<<" angle2 = "<< angle2 * transformtoAngle<<" ";

                double stream2 = 374 * cos(angle2) * 0.05 * 1.5 * 1.5;
                double t3 = sqrt(pow((19.723314 + t2 * cos(angle2) - 98),2) + pow((t2 * sin(angle2) + stream2),2));
                double angle3 = atan((t2 * sin(angle2) + stream2) / (19.723314 + t2 * cos(angle2) - 98));
                //cout<<"t3 = "<<t3<<" angle3 = "<< angle3 * transformtoAngle<<" ";

                double stream3 = 374 * cos(angle3) * 0.05 * 1.5 * 1.5;
                double t4 = sqrt(pow((19.723314 + t3 * cos(angle3) - 98),2) + pow((t3 * sin(angle3) + stream3),2));
                double angle4 = atan((t3 * sin(angle3) + stream3) / (19.723314 + t3 * cos(angle3) - 98));
                //cout<<"t4 = "<<t4<<" angle4 = "<< angle4 * transformtoAngle<<" ";

                double stream4 = 374 * cos(angle4) * 0.05 * 1.5 * 1.5;
                double t5 = sqrt(pow((19.723314 + t4 * cos(angle4) - 98),2) + pow((t4 * sin(angle4) + stream4),2));
                double angle5 = atan((t4 * sin(angle4) + stream4) / (19.723314 + t4 * cos(angle4) - 98));
                //cout<<"t5 = "<<t5<<" angle5 = "<< angle5 * transformtoAngle<<" ";

                double stream5 = 374 * cos(angle5) * 0.3 * 1.5 * 1.5;
                double t6 = sqrt(pow((19.723314 + t5 * cos(angle5) - 980 - 11760),2) + pow((t5 * sin(angle5) + stream5),2));
                double angle6 = atan((t5 * sin(angle5) + stream5) / (19.723314 + t5 * cos(angle5) - 980 - m * 9.8));
                //cout<<"t6 = "<<t6<<" angle6 = "<< angle6 * transformtoAngle;
                //cout<<" angle7 = "<<(angle6 - angle5) * transformtoAngle<<endl;
                double stream7 = 374 * cos(angle6) * 0.3 * 1.5 * 1.5;
                double t7 = sqrt(pow((t6 * cos(angle6) + 1.025 * 1000 * 9.8 * 3.2 * x /7900 - 3.2 * x * 9.8),2) + pow((t6 * sin(angle6) + stream7),2));
                double angle7 = atan((t6 * cos(angle6) + 1.025 * 1000 * 9.8 * 3.2 * x /7900 - 3.2 * x * 9.8)/(t6 * sin(angle6) + stream7));

                double h1 = cos(angle1);
                double h2 = cos(angle2);
                double h3 = cos(angle3);
                double h4 = cos(angle4);
                double h5 = cos(angle5);
                double totalheight = h1 + h2 + h3 + h4 + h5 + x * cos(angle6);
                //cout<<"total height = "<<totalheight<<endl;

                double w1 = sin(angle1);
                double w2 = sin(angle2);
                double w3 = sin(angle3);
                double w4 = sin(angle4);
                double w5 = sin(angle5);
                double totalweight = w1 + w2 + w3 + w4 + w5 + x * sin(angle6);
                //cout<<"total weight = "<<totalweight<<endl;
                if(angle7 > 0 && angle7 < 16 / transformtoAngle)
                {
                    if(angle5 > 0 && angle5 < 5 / transformtoAngle)
                    {
                        cout<<"h = "<<h[i]<<" x = "<< x <<endl;
                    }
                }
            }
        }
    }
}
